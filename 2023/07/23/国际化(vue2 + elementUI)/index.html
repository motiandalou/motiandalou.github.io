国际化(vue2 + elementUI)

2023-07-23 Vue2 2.3k 字

第一步: 安装

先下载国际化的包:注意版本,可以去 npm 社区 直接下载对应的版本,最直接的办法,找下载数量最多的版本

npm i vue-i18n@8.28.2

第二步: 配置及使用

在 src 目录新建 i18n 目录，en.json 为英文语言包，zh.json 为中文语言包,，tw.json 为繁体语言包等等语言

i18n 目录下(即文件包同级目录) 新建 index.js 文件

import Vue from "vue";
import VueI18n from "vue-i18n";
// element-ui lang导入Element的语言包 英文
import elementEnLocale from "element-ui/lib/locale/lang/en";
// element-ui lang g导入Element的语言包 中文
import elementZhLocale from "element-ui/lib/locale/lang/zh-CN";
// 导入项目中用到的英文语言包
import enLocale from "./en";
// 导入项目中用到的中文语言包
import zhLocale from "./zh";
Vue.use(VueI18n);
const messages = {
  en: {
    ...enLocale,
    ...elementEnLocale
  },
  zh: {
    ...zhLocale,
    ...elementZhLocale
  }
};

const i18n = new VueI18n({
  // 当前语种
  locale: localStorage.getItem("language") || "zh",
  // 本地语言包（'en','zh'...）
  messages,
  // 如果当前语种不存在时，默认设置当前语种
  fallbackLocale: "zh",
  // 设置 数字格式化
  numberFormats: {
    //设置 数字本地化
    en: {
      currency: {
        //添加 $
        style: "currency",
        currency: "USD"
      }
    },
    zh: {
      currency: {
        //添加 ￥
        style: "currency",
        currency: "JPY",
        currencyDisplay: "symbol"
      }
    }
  },
  // 设置 时间日期格式化
  dateTimeFormats: {
    //设置 日期时间本地化
    en: {
      short: {
        year: "numeric",
        month: "short",
        day: "numeric"
      },
      long: {
        year: "numeric",
        month: "short",
        day: "numeric",
        weekday: "short",
        hour: "numeric",
        minute: "numeric"
      }
    },
    zh: {
      short: {
        year: "numeric",
        month: "short",
        day: "numeric"
      },
      long: {
        year: "numeric",
        month: "short",
        day: "numeric",
        weekday: "short",
        hour: "numeric",
        minute: "numeric"
      }
    }
  }
});

export default i18n;


在主入口文件 main.js 中,引入上一步的 i18n 文件夹里面的 index.js 文件

import Vue from "vue";
import App from "./App";
import router from "./router";
// 引入公共样式
import "@/assets/css/common.css";
// 引入全局样式
import "@/assets/css/cover.less";
// 引入elementUI
import ElementUI from "element-ui";
//导入配置好的国际化语言包
import { i18n, loadLanguageAsync } from "./i18n";

// 阻止 vue 在启动时生成生产提示
Vue.config.productionTip = false;

// 通过插件的形式挂载，通过全局方法 Vue.use() 使用插件
Vue.use(ElementUI);

new Vue({
  el: "#app",
  router,
  components: { App },
  // 注入 配置好的国际化语言配置
  i18n,
  template: "<App/>"
});

// 路由拦截
router.beforeEach((to, from, next) => {
  // to.params.lang 路由拿到语言的值
  const lang = to.params.lang || "zh";
  loadLanguageAsync(lang).then(() => next());
});

动态加载语言包: 一次加载所有的语言包是没有必要的，特别是语言包过多的情况下，影响首屏加载速度,官网上也是给了解决方式的。

原理: 默认预装一个语言,比如中文,当切换本地存储的语言类型的时候,再动态加载对应的文件夹里面的语言包文件

// 官网链接: https://kazupon.github.io/vue-i18n/zh/guide/lazy-loading.html


import Vue from "vue";
import VueI18n from "vue-i18n";
// element-ui lang导入Element的语言包 英文
import elementEnLocale from "element-ui/lib/locale/lang/en";
// element-ui lang g导入Element的语言包 中文
import elementZhLocale from "element-ui/lib/locale/lang/zh-CN";
// element-ui lang g导入Element的语言包 繁体
import elementTwLocale from "element-ui/lib/locale/lang/zh-TW";

// 导入项目中用到的 英文 语言包
import enLocale from "./lang/en.json";
// 导入项目中用到的 中文 语言包
import zhLocale from "./lang/zh.json";
// 导入项目中用到的 繁体 语言包
import twLocale from "./lang/tw.json";

// 引入axios
import axios from "axios";

// 通过插件的形式挂载，通过全局方法 Vue.use() 使用插件
Vue.use(VueI18n);

const messages = {
  en: {
    ...enLocale,
    ...elementEnLocale
  },
  zh: {
    ...zhLocale,
    ...elementZhLocale
  },
  tw: {
    ...twLocale,
    ...elementTwLocale
  }
};
// 预装默认语言
const loadedLanguages = ["zh"];

export const i18n = new VueI18n({
  // 当前语种
  locale: localStorage.getItem("language") || "zh",
  // 本地语言包（'en','zh','tw'...）
  messages,
  // 如果当前语种不存在时，默认设置当前语种
  fallbackLocale: "zh",
  // 设置 数字格式化
  numberFormats: {
    //设置 数字本地化
    en: {
      currency: {
        //添加 $
        style: "currency",
        currency: "USD"
      }
    },
    zh: {
      currency: {
        //添加 ￥
        style: "currency",
        currency: "JPY",
        currencyDisplay: "symbol"
      }
    }
  },
  // 设置 时间日期格式化
  dateTimeFormats: {
    //设置 日期时间本地化
    en: {
      short: {
        year: "numeric",
        month: "short",
        day: "numeric"
      },
      long: {
        year: "numeric",
        month: "short",
        day: "numeric",
        weekday: "short",
        hour: "numeric",
        minute: "numeric"
      }
    },
    zh: {
      short: {
        year: "numeric",
        month: "short",
        day: "numeric"
      },
      long: {
        year: "numeric",
        month: "short",
        day: "numeric",
        weekday: "short",
        hour: "numeric",
        minute: "numeric"
      }
    }
  }
});

// 设置国际化--语言
function setI18nLanguage(lang) {
  i18n.locale = lang;
  // 设置axios请求头
  axios.defaults.headers.common["Accept-Language"] = lang;
  document.querySelector("html").setAttribute("lang", lang);
  return lang;
}

export function loadLanguageAsync(lang) {
  // 如果语言相同
  if (i18n.locale === lang) {
    return Promise.resolve(setI18nLanguage(lang));
  }

  // 如果语言已经加载
  if (loadedLanguages.includes(lang)) {
    return Promise.resolve(setI18nLanguage(lang));
  }

  // 根据路由传进来的值--lang(设置的语言)--如果尚未加载语言
  return import(`./lang/${lang}.json`).then(messages => {
    // 会重设框架的语言环境，如果在调用这个方法之前语言环境里面有文件那么会覆盖这个文件
    i18n.setLocaleMessage(lang, messages.default);
    // 加入--预装默认语言
    loadedLanguages.push(lang);
    // 设置国际化--语言
    return setI18nLanguage(lang);
  });
}
